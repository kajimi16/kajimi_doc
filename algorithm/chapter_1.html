eps

由于浮点数的精度误差，需要一个范围判断两数相等 经验表明1e-8差不多

const double eps = 1e-8;
#define Equ(a,b)     (((fabs((a)-(b)))) < (eps))  //等于
#define More(a,b)    (((a) - (b)) > ( eps))       //大于
#define Less(a,b)    (((a) - (b)) < (-eps))       //小于
#define MoreEqu(a,b) (((a) - (b)) > (-eps))       //大等于
#define LessEqu(a,b) (((a) - (b)) < ( eps))       //小等于
