int binary_search(int arr[], int n, int left, int right)
{
    int mid = left + (right - left) / 2;
    while (left <= right)
    {
        mid = left + (right - left) / 2;
        if (arr[mid] == n)
        {
            return mid;
        }
        else if (arr[mid] > n)
        {
            right = mid -1;
        }
        else
        {
            left = mid + 1;
        }
    }
    
    return -1;
}

typedef long long LL
LL binaryPow(LL a, LL b, LL m)
{
    LL ans = 1;
    while (b > 0)
    {
        if (b > 0)
        {
            if (b & 1)
            {
                ans = ans * a % m;
            }
            a = a * a % m;
            b >>= 1;
        }
    }
    return ans;
}
